# 10x10 grid size
width = 10
height = 10

# Circle center and radius
h, k = 5, 5 # Circle center (h, k)
r = 4 # Circle radius

# Loop through each point in the grid
for y in range(height):
for x in range(width):
# Check if the point is within the circle using the equation
if (x - h) ** 2 + (y - k) ** 2 <= r ** 2:
print("*", end="") # Print "*" if within the circle
else:
print(" ", end="") # Print space if outside the circle

from browser import document, html, window

# 畫布設置
canvas = html.CANVAS(width=300, height=300)
document <= canvas
ctx = canvas.getContext('2d')

# 方格的大小
grid_size = 30
width = 10 # 方格數量
height = 10

# 圓心與半徑
h, k = 5, 5
r = 4

# 繪製方格
def draw_grid():
ctx.strokeStyle = "gray"
for x in range(0, width * grid_size, grid_size):
for y in range(0, height * grid_size, grid_size):
ctx.strokeRect(x, y, grid_size, grid_size)

# 繪製圓形邊界
def draw_circle():
ctx.fillStyle = "black"
for x in range(width):
for y in range(height):
# 根據圓的公式檢查每個點是否在圓內
if (x - h) ** 2 + (y - k) ** 2 <= r ** 2:
# 計算方格的 x, y 位移
x_pos = x * grid_size
y_pos = y * grid_size
ctx.fillRect(x_pos, y_pos, grid_size, grid_size)

# 初始化畫面
draw_grid()
draw_circle()
